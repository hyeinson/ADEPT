fig, ax = plt.subplots(1,1, figsize = (5,1.8), dpi = 200)

sns.lineplot(data = df, marker = 'o', markersize = 7,
             x = 'Hours', y = 'Green (a.u.)', ax = ax, color = '#14A201', lw = 2)
right_ax = ax.twinx()
sns.lineplot(data = df, marker = 'o', markersize = 7, 
                 x = 'Hours', y = 'Red (a.u.)', ax = right_ax, color = '#A31800', lw = 2)
